### 一

```
async function async1() {
  console.log("async1 start");
  console.log(await async2());
  console.log("async1 end");
}
async function async2() {
  console.log("async2");
}

console.log("script start");

setTimeout(function () {
  console.log("setTimeout");
}, 0);
async1();

new Promise((resolve, reject) => {
  console.log("promise1");
  resolve();
}).then(() => {
  console.log("promise2");
});

console.log("script end");
```

### 二

改造 async2 function

```
async function async2() {
  console.log("async2");
  return "heihei";
}
```

### 三

继续改造 async2 function

```
async function async2() {
  console.log("async2");
  return Promise.resolve("heihei");
}
```

### 四

继续改造 async2 function

```
async function async2() {
  console.log("async2");
  const thenable = {
    then(resolve, reject) {
      console.log("thenable obj in async2 result");
      resolve("lala");
    },
  };
  return thenable;
}
```

### 总结

1. 第一种情况还比较正常，能够简单推测到正确结果
2. 执行完 async1 作用域中的 await async2() 后，会立即把 async2 默认返回的 Promise 挂入微任务队列，阻塞 await async2()  之后的代码，包括包裹 await async2()  的 console 行为
3. 三和四都在 async2 中返回了一个 thenable object，而 then 返回结果的处理又需要挂载到微任务队列，因此 then 的返回结果会在 promise 2 之后打印

情况四的代码，其实和下面代码的结果是一致的

```
async function async1() {
  console.log("async1 start");
  console.log(
    await Promise.resolve(
      new Promise((resolve) => {
        console.log("async 2");
        resolve();
      }).then((res) => {
        console.log("thenable in async2 result");
        return "haha";
      })
    )
  );
  console.log("async1 end");
}

console.log("script start");

setTimeout(function () {
  console.log("setTimeout");
}, 0);
async1();

new Promise((resolve, reject) => {
  console.log("promise1");
  resolve();
}).then(() => {
  console.log("promise2");
});

console.log("script end");
```
